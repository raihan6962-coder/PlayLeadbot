var sheet = ss.getSheetByName(tab); if (!sheet) sheet = ss.insertSheet(tab); var row = data.row; if (sheet.getLastRow() === 0) { sheet.appendRow(Object.keys(row)); } sheet.appendRow(Object.values(row)); if (sCol > 0) qSheet.getRange(j+1, sCol).setValue("Yes"); break; } } } var sentSheet = ss.getSheetByName("Email Sent"); if (!sentSheet) sentSheet = ss.insertSheet("Email Sent"); if (sentSheet.getLastRow() === 0) { sentSheet.appendRow(["App ID","App Name","Email","Sent At"]); } sentSheet.appendRow([data.app_id, data.app_name||"", data.email||"", new Date().toLocaleString()]); } else if (action === "get_pending") { var allSheet = ss.getSheetByName("All Leads"); if (!allSheet) return ContentService.createTextOutput(JSON.stringify({status:"ok",leads:[]})).setMimeType(ContentService.MimeType.JSON); var vals = allSheet.getDataRange().getValues(); if (vals.length < 2) return ContentService.createTextOutput(JSON.stringify({status:"ok",leads:[]})).setMimeType(ContentService.MimeType.JSON); var headers = vals[0]; var colIdx = {}; headers.forEach(function(h, i) { colIdx[h] = i; }); var leads = []; for (var i = 1; i < vals.length; i++) { var row = vals[i]; var emailSent = String(row[colIdx["Email Sent"]] || "").trim().toLowerCase(); if (emailSent === "no" || emailSent === "") { leads.push({ app_id: String(row[colIdx["App ID"]]||""), app_name: String(row[colIdx["App Name"]]||""), developer: String(row[colIdx["Developer"]]||""), email: String(row[colIdx["Email"]]||""), category: String(row[colIdx["Category"]]||""), installs: Number(row[colIdx["Installs"]]||0), score: row[colIdx["Score"]] ? Number(row[colIdx["Score"]]) : null, url: String(row[colIdx["URL"]]||""), keyword: String(row[colIdx["Keyword"]]||""), }); } } return ContentService.createTextOutput(JSON.stringify({status:"ok",leads:leads})).setMimeType(ContentService.MimeType.JSON); var records = []; for (var i = 1; i < vals.length; i++) { var rec = {}; headers.forEach(function(h, j) { rec[h] = String(vals[i][j]||""); }); records.push(rec); } return ContentService.createTextOutput(JSON.stringify({status:"ok",records:records})).setMimeType(ContentService.MimeType.JSON);
